[9. 回文数](https://leetcode.cn/problems/palindrome-number/)

```javascript
// 方法一
/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function (x) {
  return x.toString() === x.toString().split('').reverse().join('')
}

// 方法二：不将整数转为字符串
/**
 * @param {number} x
 * @return {boolean}
 */
var isPalindrome = function (x) {
  if (x < 0) return false
  let n = 10 ** Math.floor(Math.log10(x))
  while (n > 1 && x > 0) {
    if (Math.floor(x / n) !== x % 10) return false
    x = Math.floor((x % n) / 10)
    n /= 100
  }
  return true
}
```

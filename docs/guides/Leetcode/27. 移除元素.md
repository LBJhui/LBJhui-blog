/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function (nums, val) {
  let count = 0
  nums.map((item, index) => {
    item === val ? delete nums[index] : count++
  })
  nums.sort((a, b) => a - b)
  return count
}
